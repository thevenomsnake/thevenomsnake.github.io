### 进阶练习3

```bash
#include <stdio.h>
int main(int argc, const char * argv[]) {
    char a, b, c;
    int sum;
    scanf("%c%c%c",&a,&b,&c);
    sum = (a-'0')*100+(b-'0')*10+(c-'0');
    printf("the number is %d\n",sum);
    return 0;
}
```

### 进阶练习11威力加强（偷懒）版

```bash
#include <stdio.h>
int main(int argc, const char * argv[]) {
    char a[1000];
    int i;
    gets(a);
    puts(a);
    printf("\n");
    for(i=0;a[i]!='\0';i++)
    {
        if(a[i]>='A'&&a[i]<='V')
            a[i]+=4;
        else if(a[i]>='a'&&a[i]<='v')
            a[i]+=4;
    }
    for (i=0;a[i]!='\0';i++)
    {
        switch(a[i]){
            case 'w':
                a[i] = 'a';
                break;
            case 'x':
                a[i] = 'b';
                break;
            case 'y':
                a[i] = 'c';
                break;
            case 'z':
                a[i] = 'd';
                break;
            case 'W':
                a[i] = 'A';
                break;
            case 'X':
                a[i] = 'B';
                break;
            case 'Y':
                a[i] = 'C';
                break;
            case 'Z':
                a[i] = 'D';
                break;           
        }       
    }
    for (i=0;a[i]!='\0';i++)
    {
        printf("%c",a[i]);
    }
    printf("\n");
    return 0;
}
```
